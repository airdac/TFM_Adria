library(Rdimtools)
library(dimRed)
library(microbenchmark)

set.seed(42)

file_name <- "MDS_benchmark_R_lib.csv"
save_benchmark <- function(mbm, n_obs, d) {
  mbm_df <- as.data.frame(mbm)
  colnames(mbm_df) <- c("method", "time_ns")
  mbm_df$n_obs <- n_obs
  mbm_df$d <- d
  if (!file.exists(file_name)) {
    # If it doesnât exist, write with headers
    write.csv(mbm_df, file_name, row.names = FALSE)
  } else {
    # If it exists, append without writing headers again
    write.table(mbm_df, file_name, row.names = FALSE, col.names = FALSE, sep = ",", append = TRUE)
  }
}

The goal of this analysis is to test the limits of \(\texttt{Rdimtools::do.mds}\). Specifically, we will run Rdimtoolsâs implementation of Classical MDS on datasets of increasing number of observations and dimensionality in order to compare its limitations as well as its complexity. We will also compare it with \(\texttt{dimRed::embed(.method="MDS", ...)}\).

We will first run the function on the iris dataset to understand it better.

Performance on the iris dataset

# load iris dataset
data(iris)
x <- as.matrix(iris[, 1:4])
lab <- as.factor(iris[, 5])
Rdimtools.iris <- do.mds(x, ndim = 2)
dimRed.iris <- embed(x, "MDS", ndim=2)
dimRed.iris <- as.data.frame(dimRed.iris@data@data)
plot(Rdimtools.iris$Y, pch = 19, col = lab, main = "Rdimtools on Iris")

plot(dimRed.iris, pch=19, col=lab, main = "dimRed on Iris")

mbm <- microbenchmark(
  Rdimtools.iris = do.mds(x, ndim = 2)
  , times = 100  # Number of iterations
)
## Warning in microbenchmark(Rdimtools.iris = do.mds(x, ndim = 2), times = 100):
## less accurate nanosecond times to avoid potential integer overflows
save_benchmark(mbm, n_obs = nrow(iris), d = 4)

mbm <- microbenchmark(
  dimRed.iris = do.mds(x, ndim = 2)
  , times = 100  # Number of iterations
)
save_benchmark(mbm, n_obs = nrow(iris), d = 4)
